function add_custom_code() {
    if (!is_user_logged_in() || !current_user_can('administrator')) {
        echo '<script async src="https://pagead2.googlesyndication.com/pagead/js/adsbygoogle.js?client=ca-pub-9156332944666352" crossorigin="anonymous"></script>';
    }
}
add_action('wp_head', 'add_custom_code');
